## 2944 -  Sort an Array

=== "bubble sort(TLE)"

    ```c++
    // bubble sort
    
    class Solution {
    public:
        vector<int> sortArray(vector<int>& nums) {
            int tmp=0;
            int size = nums.size();
            for(int i=0; i<size-1; ++i)
            {
                for(int j=0; j<size-1-i; ++j)
                {
                    if(nums[j] > nums[j+1])
                    {
                        tmp = nums[j];
                        nums[j] = nums[j+1];
                        nums[j+1] = tmp;
                    }
                }
            }
            return nums;
        }
    };
    ```



=== "selection sort(TLE)"

    ```c++
    // selection sort
    class Solution {
    public:
        vector<int> sortArray(vector<int>& nums) {
            int size = nums.size();
            int tmp=0;
            for(int i=0; i<size; ++i)
            {
                int maxVal = nums[0];
                int maxIdx = i;
                for(int j=i; j<size; ++j)
                {
                    if(nums[j] < maxVal)
                    {
                        maxVal = nums[j];
                        maxIdx = j;
                    }
                }
    
                tmp = nums[maxIdx];
                nums[maxIdx] = nums[i];
                nums[i] = tmp;
            }
    
            return nums;
        }
    };
    ```



=== "insertion sort(TLE)"

    ```c++
    // insertion sort
    class Solution {
    public:
        vector<int> sortArray(vector<int>& nums) {
            int tmp=0;
            int size = nums.size();
            for(int i=1; i<size; ++i)
            {
                int idx = i;
                while(idx > 0)
                {
                    if(nums[idx] < nums[idx-1])
                    {
                        tmp = nums[idx];
                        nums[idx] = nums[idx-1];
                        nums[idx-1] = tmp;
                    }else
                    {
                        break;
                    }
                    idx--;
                }
            }
            return nums;
        }
    };
    ```
